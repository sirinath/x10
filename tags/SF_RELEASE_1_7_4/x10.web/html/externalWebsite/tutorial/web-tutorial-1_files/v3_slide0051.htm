Futures can deadlock
nullable future<int> f1=null;
nullable future<int> f2=null;
void main(String[] args) {
  f1 = future(here){a1()};
  f2 = future(here){a2()};
  f1.force();
}
int a1() {
  nullable future<int> tmp=null;
  do {
    tmp=f2;
  } while (tmp == null);
  return tmp.force();
}
int a2() {
  nullable future<int> tmp=null;
  do {
    tmp=f1;
  } while (tmp == null);
  return tmp.force();
}
cyclic wait condition
