%%FILE-NOT-USED%% \subsection{Place constraints}
%%FILE-NOT-USED%% \label{PlaceTypes}
%%FILE-NOT-USED%% \label{PlaceType}
%%FILE-NOT-USED%% \index{place types}
%%FILE-NOT-USED%% \label{DepType:PlaceType}\index{placetype}
%%FILE-NOT-USED%% 
%%FILE-NOT-USED%% An \Xten{} computation spans multiple places
%%FILE-NOT-USED%% (\Sref{XtenPlaces}). Each place constains data and activities that
%%FILE-NOT-USED%% operate on that data.  \XtenCurrVer{} does not permit the dynamic
%%FILE-NOT-USED%% creation of a place. Each \Xten{} computation is initiated with a
%%FILE-NOT-USED%% fixed number of places, as determined by a configuration parameter.
%%FILE-NOT-USED%% In this section we discuss how the programmer may supply place type
%%FILE-NOT-USED%% information, thereby allowing the compiler to check data locality,
%%FILE-NOT-USED%% i.e., that data items being accessed in an atomic section are local.
%%FILE-NOT-USED%% 
%%FILE-NOT-USED%% \begin{grammar}
%%FILE-NOT-USED%% PlaceConstraint     \: \xcd"!" Place\opt \\
%%FILE-NOT-USED%% Place              \:   Expression \\
%%FILE-NOT-USED%% \end{grammar}
%%FILE-NOT-USED%% 
%%FILE-NOT-USED%% Because of the importance of places in the \Xten{} design, special
%%FILE-NOT-USED%% syntactic support is provided for constrained types involving places.
%%FILE-NOT-USED%% 
%%FILE-NOT-USED%% All \Xten{} classes extend the class
%%FILE-NOT-USED%% \xcd"x10.lang.Object", which defines a property
%%FILE-NOT-USED%% \xcd"home" of type
%%FILE-NOT-USED%% \xcd"Place".
%%FILE-NOT-USED%% 
%%FILE-NOT-USED%% If a constrained reference type \xcd"T" has an \xcd"!p" suffix,
%%FILE-NOT-USED%% the constraint for \xcd"T" is implicitly assumed to contain the clause
%%FILE-NOT-USED%% \xcd"self.home==p"; that is,
%%FILE-NOT-USED%% \xcd"C{c}!p" is equivalent to \xcd"C{self.home==p && c}".
%%FILE-NOT-USED%% 
%%FILE-NOT-USED%% The place \xcd"p" may be ommitted. It defaults to \xcd"this" 
%%FILE-NOT-USED%% for types in field declarations, and to \xcd"here" elsewhere.
%%FILE-NOT-USED%% 
%%FILE-NOT-USED%% 
%%FILE-NOT-USED%% % The place specifier \xcd"any" specifies that the object can be
%%FILE-NOT-USED%% % located anywhere.  Thus, the location is unconstrained; that is,
%%FILE-NOT-USED%% % \xcd"C{c}!any" is equivalent to \xcd"C{c}".
%%FILE-NOT-USED%% 
%%FILE-NOT-USED%% % XXX ARRAY
%%FILE-NOT-USED%% %The place specifier \xcd"current" on an array base type
%%FILE-NOT-USED%% %specifies that an object with that type at point \xcd"p"
%%FILE-NOT-USED%% %in the array 
%%FILE-NOT-USED%% %is located at \xcd"dist(p)".  The \xcd"current" specifier can be
%%FILE-NOT-USED%% %used only with array types.
%%FILE-NOT-USED%%
